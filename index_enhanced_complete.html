🔄 数据获取


// 获取附近寄存点
fetchNearbyLockers(latitude, longitude) {
    uni.request({
        url: 'http://localhost:8000/api/lockers/distribution',
        method: 'POST',
        data: {
            city: this.currentCity,
            longitude: longitude,
            latitude: latitude,
            radius: 10
        },
        success: (res) => {
            this.handleNearbyLockersSuccess(res.data);
        }
    });
}
